method mul(a0:int, b0:int) returns (p: int)
  requires a0 >= 0 && b0 >= 0
  ensures p == a0 * b0
  {
      var a:int := a0;
      var b:int := b0;
      p := 0;
      while (a > 0)
        decreases //fill in decreases
        invariant //fill in invariant
        {
          if (a % 2 == 1)
          {
            p := p + b;
            a := a - 1;
          }
          assert a % 2 == 0;
          a := a / 2;
          b := b * 2;
        }
  }
