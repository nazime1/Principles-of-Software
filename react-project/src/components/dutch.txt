requires //fill in requires
modifies arr
ensures 0 <= k <= arr.Length
ensures //fill in ensures
{
	k := 0;
	var j := arr.Length;
	while (k < j) 
	invariant //fill in invariant
	decreases //fill in decreases
	{
		if (arr[k] == 'r') {
			k := k + 1;
